	if(x==0){return 0;}
	if(x==1){return 1;}
	else{
		return (x%10)+2*binarios(x/10);}}

void insertar_elemento(int A[],int& tl, int indice){
for(int i=tl;i>indice;i--){
A[i]=A[i-1];}
A[indice+1]++;
tl++;}
inserta el ultimo elemento en tl ya que si tl era 5 el ultimo elemento del arreglo estaba en A[4], ahora esta en 5, asi hasta encontrar el indice una vez ahi inserta el elemento en el indice+1 el cual es una copia de A[indice] finalmente actualiza el tl
 void borrar_elemento(int A[],int& tl, int indice){
		for(int i=indice;i<tl;i++){
		A[i]=A[i+1];}
		tl--;}
 al borrar el elemento arrancamos del indice o encontramos el elemento y sobrescribimos todos los A[i] con el valor de A[i+1] y finalmente decrementamos el tl
 void borrar_rango(int A[],int& tl, int inf,int sup){ 
//borra un rango de elementos incluidos los extremos
	if(inf==sup){borrar_elemento(A,tl,inf);}
	else{
	for(int i=inf;i + (sup - inf + 1) <tl;i++){
	A[i]=A[i+(sup-inf)+1];}
tl-=(sup-inf)+1;}}
